#include <bits/stdc++.h>

using namespace std;

int main(){
    ios::sync_with_stdio(false);cin.tie(0);
    int n,m;cin>>n>>m;
    char a[n][m];
    for(int i=0;i<n;i++){
        for(int j=0;j<m;j++){
            cin>>a[i][j];
        }
    }
    int y,x;cin>>y>>x;
    x-=1;
    y-=1;
    string s;cin>>s;
    for(int i=0;i<s.size();i++){
        if(s[i]=='L'){
            if(x!=0){
                x-=1;
                cout<<a[y][x];
            }
        }else if(s[i]=='R'){
            if(x!=s.size()-1){
                x+=1;
                cout<<a[y][x];
            }
        }else if(s[i]=='U'){
            if(y!=0){
                y-=1;
                cout<<a[y][x];
            }
        }else if(s[i]=='D'){
            if(y!=s.size()-1){
                y+=1;
                cout<<a[y][x];
            }
        }
    }
    cout<<endl;
    return 0;
}
